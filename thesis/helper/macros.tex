%My macros for making it easier to draw graphs 


% For 45째 line -> center + 0.21
% For 22.5째 line -> center.x + 0.27 , center.y + 0.11
% Cluster attacks -> cluster.y + 0.21 for top and - 0.21 for bottom
\newcommand{\DrawSelfAttackLeftTopCluster}[2]{% X, Y
\draw[-{To[length=4, width=5]}, line width=0.3mm] (#1, #2) 
.. controls (#1 -0.35, #2 + 0.34) and (#1 - 0.75, #2 - 0.12) .. 
(#1 - 0.13, #2 - 0.12);
}

\newcommand{\DrawSelfAttackRightTopCluster}[2]{% X, Y
\draw[-{To[length=4, width=5]}, line width=0.3mm] 
            (#1, #2) .. controls 
            (#1 + 0.35, #2 + 0.34) and 
            (#1 + 0.75, #2 - 0.12) .. 
            (#1 + 0.13, #2 - 0.12);
}

\newcommand{\DrawSelfAttackRightSingleton}[2]{% X, Y
\def \xShift{0.12}
\def \yShift{0.15}
\draw[-{To[length=4, width=5]}, line width=0.3mm]
            (#1 + \xShift + 0.05, #2 + \yShift + 0.1) .. controls
            (#1 + \xShift + 0.28, #2 + \yShift + 0.68) and
            (#1 + \xShift + 0.75, #2 + \yShift) ..
            (#1 + \xShift + 0.13, #2 + \yShift + 0.02);
}

\newcommand{\DrawSelfAttackLeftSingleton}[2]{% X, Y
\def \xShift{0.12}
\def \yShift{0.15}
\draw[-{To[length=4, width=5]}, line width=0.3mm]
            (#1 - \xShift - 0.05, #2 + \yShift + 0.1) .. controls
            (#1 - \xShift - 0.28, #2 + \yShift + 0.68) and
            (#1 - \xShift - 0.75, #2 + \yShift) ..
            (#1 - \xShift - 0.13, #2 + \yShift + 0.02);
}

\newcommand{\DrawAttackHorizontal}[5]{%Direction L.x L.y R.x R.y 
% R = Right to Left
% L = Left to Right
\ifthenelse{\equal{#1}{R}}{
\draw[-{To[length=4, width=5]}, line width=0.3mm] (#2 + 0.3,#3) -- (#4 - 0.3 , #5);
}{
\ifthenelse{\equal{#1}{L}}{
\draw[-{To[length=4, width=5]}, line width=0.3mm]  (#2 - 0.3,#3) -- (#4 + 0.3 , #5);
}{
\ifthenelse{\equal{#1}{B}}{
\draw[-{To[length=4, width=5]}, line width=0.3mm]  (#2 - 0.3,#3 + 0.08) -- (#4 + 0.3 , #5 + 0.08);
\draw[{To[length=4, width=5]}-, line width=0.3mm]  (#2 - 0.3,#3 - 0.08) -- (#4 + 0.3 , #5 - 0.08);
}{}
}}}
